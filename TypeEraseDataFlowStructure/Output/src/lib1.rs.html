use compute::Output;
#[derive(Debug)]
pub struct OutputImpl {

}
impl Output for OutputImpl {
    fn new() -> OutputImpl {
        OutputImpl {}
    }
    fn do_output(&self, name: &str, lines: usize) {
        print!("\n  file {:?} has {} lines of code", name, lines);
    }
}
