mod file_utilities;
use file_utilities::open_file_for_read;

#[derive(Debug)]
pub struct Input {
    name: String,
}
impl Input {
    pub fn new() -> Input {
        Input {
            name: String::new(),
        }
    }
    pub fn do_input(&mut self, name: &str) -> Option<File> {
        let rslt = open_file_for_read(name);
        if let Ok(file) = rslt {
            return Some(file);
        }
        else {
            print!("\n  can't open file {:?}", name);
            return None;
        }
    }
}
