use compute::Output;
#[derive(Debug)]
pub struct OutputImpl {

}
impl Output for OutputImpl {
    fn do_output(&self, name: &str, lines: usize) {
        print!("\n  {} lines in file {:?}", lines, name);
    }
}
impl OutputImpl {
    pub fn new() -> OutputImpl {
        OutputImpl {}
    }
}
#[cfg(test)]
mod tests {
    #[test]
    fn it_works() {
        assert_eq!(2 + 2, 4);
    }
}
